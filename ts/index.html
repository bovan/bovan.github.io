Opprinnelig TypeScript

                
                    class LogOnClick {
                        constructor(el: HTMLElement) {
                            el.addEventListener(
                                "click", 
                                (e) => this.log(el)
                            );
                        }
                        log(el) {
                            console.log("Du klikket på", el);
                        }
                    }

Kompilert javaScript

                
                    var LogOnClick = /** @class */ (function () {
                        function LogOnClick(el) {
                            var _this = this;
                            el.addEventListener(
                                "click", 
                                function (e) { return _this.log(el); })
                            ;
                        }
                        LogOnClick.prototype.log = function (el) {
                            console.log("Du klikket på", el);
                        };
                        return LogOnClick;
                    }());
